% ── Custom commands ───────────────────────────────────────────────────────────

% Section header: semibold blue, 11pt
\newcommand{\rsection}[1]{%
  {\fontsize{11}{14}\selectfont\bfseries\color{accent}#1}%
}

% Right sidebar section header (uses \sidebarheadingsep after)
\newcommand{\rsidebarsection}[1]{%
  \rsection{#1}\par\nointerlineskip\vspace{\sidebarheadingsep}%
}

% Left column section header (uses \sectiontocontent after)
\newcommand{\rleftsection}[1]{%
  \rsection{#1}\par\nointerlineskip\vspace{\sectiontocontent}%
}

% Subsection header (right column): semibold dark, 9pt
\newcommand{\rsubsection}[1]{%
  {\fontsize{9}{12}\selectfont\bfseries\color{heading}#1}%
  \par\nointerlineskip\vspace{\sidebarsubheadingsep}%
}

% Body text environment (8pt / 11pt leading)
\newcommand{\bodytext}{%
  \fontsize{8}{11}\selectfont\color{body}%
}

% Date text: 7pt, muted
\newcommand{\datetext}[1]{%
  {\fontsize{7}{9}\selectfont\color{muted}#1}%
}

% Job entry: title (regular) · company (semibold)  date
\newcommand{\jobentry}[3]{%
  {\fontsize{9}{12}\selectfont\color{heading}%
    #1 \,\textperiodcentered\, \textbf{#2}}%
  \hspace{5pt}\datetext{#3}%
  \par\nointerlineskip\vspace{\jobrolesep}%
}

% Sub-role (e.g. promotion): title (regular)  date
\newcommand{\subrole}[2]{%
  {\fontsize{9}{12}\selectfont\color{heading}#1}%
  \hspace{5pt}\datetext{#2}%
  \par\nointerlineskip\vspace{\jobrolesep}%
}

% Last role before bullets (uses roletodesc gap)
\newcommand{\lastrole}[2]{%
  {\fontsize{9}{12}\selectfont\color{heading}#1}%
  \hspace{5pt}\datetext{#2}%
  \par\nointerlineskip\vspace{\roletodesc}%
}

% Single job entry (no sub-roles, goes straight to bullets)
\newcommand{\singlejobentry}[3]{%
  {\fontsize{9}{12}\selectfont\color{heading}%
    #1 \,\textperiodcentered\, \textbf{#2}}%
  \hspace{5pt}\datetext{#3}%
  \par\nointerlineskip\vspace{\roletodesc}%
}

% Bullet list for job descriptions
\newlist{bullets}{itemize}{1}
\setlist[bullets]{
  label=\textbullet,
  leftmargin=12pt,
  labelsep=4pt,
  itemsep=\bulletsep,
  parsep=0pt,
  topsep=0pt,
  partopsep=0pt,
}

% Spacing between job blocks
\newcommand{\jobsep}{\par\nointerlineskip\vspace{\jobseplen}}

% Spacing between project blocks
\newcommand{\projectsep}{\par\nointerlineskip\vspace{\projectseplen}}

% Project entry: Name · Type  Year + description paragraph
\newcommand{\projectentry}[4]{%
  {\fontsize{9}{12}\selectfont\color{heading}%
    \textbf{#1} \,\textperiodcentered\, #2}%
  \hspace{5pt}\datetext{#3}%
  \par\nointerlineskip\vspace{\projectdesc}%
  {\bodytext #4\par}%
}
